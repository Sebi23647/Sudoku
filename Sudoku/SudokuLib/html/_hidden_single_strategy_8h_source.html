72// Hidden single: finds a number n that can go in exactly one empty cell in a unit.
73class HiddenSingleStrategy : public IHintStrategy {
74public:
75 std::optional<Hint> getHint(const HintContext& ctx) const override {
76 // Row-based
77 for (int r = 0; r < 9; ++r) {
78 for (int n = 1; n <= 9; ++n) {
79 int count = 0, lastC = -1;
80 for (int c = 0; c < 9; ++c) {
81 if (ctx.game.getCellState(r, c) == CellState::EMPTY) {
82 const auto& cand = ctx.candidatesAt(r, c);
83 if (std::find(cand.begin(), cand.end(), n) != cand.end()) {
84 ++count; lastC = c;
85 }
86 }
87 }
88 if (count == 1) {
89 Hint h; h.row = r; h.col = lastC; h.value = n; h.type = HintType::HiddenSingle;
90 h.message = "Hidden Single (row): cifra " + std::to_string(n) + " poate fi plasata doar la (" + std::to_string(r) + "," + std::to_string(lastC) + ")";
91 return h;
92 }
93 }
94 }
95
96 // Column-based
97 for (int c = 0; c < 9; ++c) {
98 for (int n = 1; n <= 9; ++n) {
99 int count = 0, lastR = -1;
100 for (int r = 0; r < 9; ++r) {
101 if (ctx.game.getCellState(r, c) == CellState::EMPTY) {
102 const auto& cand = ctx.candidatesAt(r, c);
103 if (std::find(cand.begin(), cand.end(), n) != cand.end()) {
104 ++count; lastR = r;
105 }
106 }
107 }
108 if (count == 1) {
109 Hint h; h.row = lastR; h.col = c; h.value = n; h.type = HintType::HiddenSingle;
110 h.message = "Hidden Single (col): cifra " + std::to_string(n) + " poate fi plasata doar la (" + std::to_string(lastR) + "," + std::to_string(c) + ")";
111 return h;
112 }
113 }
114 }
115
116 // Box-based
117 for (int br = 0; br < 3; ++br) {
118 for (int bc = 0; bc < 3; ++bc) {
119 int sr = br * 3, sc = bc * 3;
120 for (int n = 1; n <= 9; ++n) {
121 int count = 0, lastR = -1, lastC = -1;
122 for (int i = 0; i < 3; ++i) {
123 for (int j = 0; j < 3; ++j) {
124 int r = sr + i, c = sc + j;
125 if (ctx.game.getCellState(r, c) == CellState::EMPTY) {
126 const auto& cand = ctx.candidatesAt(r, c);
127 if (std::find(cand.begin(), cand.end(), n) != cand.end()) {
128 ++count; lastR = r; lastC = c;
129 }
130 }
131 }
132 }
133 if (count == 1) {
134 Hint h; h.row = lastR; h.col = lastC; h.value = n; h.type = HintType::HiddenSingle;
135 h.message = "Hidden Single (box): cifra " + std::to_string(n) + " poate fi plasata doar la (" + std::to_string(lastR) + "," + std::to_string(lastC) + ")";
136 return h;
137 }
138 }
139 }
140 }
141
142 return std::nullopt;
143 }
144};
