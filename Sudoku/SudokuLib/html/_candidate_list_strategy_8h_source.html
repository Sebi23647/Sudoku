33class CandidateListStrategy : public IHintStrategy {
34public:
35 std::optional<Hint> getHint(const HintContext& ctx) const override {
36 for (int r = 0; r < 9; ++r) {
37 for (int c = 0; c < 9; ++c) {
38 if (ctx.game.getCellState(r, c) == CellState::EMPTY) {
39 const auto& candidates = ctx.candidatesAt(r, c);
40 if (!candidates.empty() && candidates.size() <= 3) {
41 Hint h; h.row = r; h.col = c; h.type = HintType::CandidateList; h.candidates = candidates;
42 h.message = "Candidati pe (" + std::to_string(r) + "," + std::to_string(c) + "): ";
43 for (size_t i = 0; i < candidates.size(); ++i) {
44 h.message += std::to_string(candidates[i]);
45 if (i + 1 < candidates.size()) h.message += ", ";
46 }
47 return h;
48 }
49 }
50 }
51 }
52 return std::nullopt;
53 }
54};
